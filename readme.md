    <h3>
    Arch Linux
    </h3>
    <code>
    sudo pacman -S git ruby gcc lolcat
    </code>
    <br>
    <h3>
    Void Linux
    </h3>
    <code>
    sudo xbps-install -S git ruby gcc lolcat 
    </code>
    <br>
    <h3>
    Ubuntu/Debian
    </h3>
    <code>
    sudo apt install git ruby gcc lolcat -y
    </code>
    <br>
    <h3>
    Fedora:
    </h3>
    <code>
    sudo dnf install git ruby gcc lolcat
    </code>
